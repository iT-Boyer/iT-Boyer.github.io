гҖҗжіЁж„ҸгҖ‘жңҖеҗҺжӣҙж–°дәҺ September 2, 2014пјҢж–ҮдёӯеҶ…е®№еҸҜиғҪе·ІиҝҮж—¶пјҢиҜ·и°Ёж…ҺдҪҝз”ЁгҖӮ

жҢүжҜ”дҫӢзј©ж”ҫ

жҢүжҜ”дҫӢзј©ж”ҫпјҢиҝҷжҳҜеңЁInterface Builderдёӯж— жі•и®ҫзҪ®зҡ„еҶ…е®№гҖӮ
иҖҢеңЁд»Јз ҒдёӯпјҢжңүеҰӮдёӢдёӨз§Қе®һзҺ°ж–№ејҸ:

  1. дҪҝз”ЁNSLayoutConstraintзұ»еһӢзҡ„еҲқе§ӢеҢ–еҮҪж•°дёӯзҡ„multiplierеҸӮж•°е°ұеҸҜд»Ҙйқһеёёз®ҖеҚ•зҡ„и®ҫзҪ®жҢүжҜ”дҫӢзј©ж”ҫгҖӮ
  2. еҗҢж—¶д№ҹеҸҜд»Ҙи®ҫзҪ®дёҚеҗҢNSLayoutAttributeеҸӮж•°жқҘиҫҫеҲ°ж„ҸжғідёҚеҲ°зҡ„ж•ҲжһңпјҢжҜ”еҰӮвҖңAзҡ„WidthзӯүдәҺBзҡ„Heightзҡ„2еҖҚвҖқиҝҷж ·зҡ„ж•ҲжһңгҖӮ

зҺ°еңЁе°ұжӢҝдёҖдёӘз®ҖеҚ•зҡ„**UIButton**еҒҡзӨәдҫӢпјҢеңЁViewControllerдёӯеҲӣе»әдёҖдёӘUIButtonеӯ—ж®өпјҡ
{%codeblock lang:objc%} UIButton *btn;
{%endcodeblock%}

йңҖжұӮ 1пјҡ

  1. иҰҒжұӮ**UIButtonж°ҙе№іеұ…дёӯпјҢе§Ӣз»Ҳи·қзҰ»зҲ¶Viewеә•йғЁ20**еҚ•дҪҚпјҢе…¶й«ҳеәҰжҳҜзҲ¶Viewй«ҳеәҰзҡ„дёүеҲҶд№ӢдёҖгҖӮ
  2. дҪҝз”ЁKVOжқҘзӣ‘жҺ§**UIButton**зҡ„еӨ§е°Ҹе№¶е®һж—¶иҫ“еҮәеҲ°еұҸе№•дёҠгҖӮ
    {%codeblock lang:objc%}

  • (void)viewDidLoad { [super viewDidLoad];

    //еҲӣе»әUIButtonпјҢдёҚйңҖиҰҒи®ҫзҪ®frame btn = [UIButton buttonWithType:UIButtonTypeRoundedRect]; [btn setTitle:@“mgen” forState:UIControlStateNormal]; btn.backgroundColor = [UIColor greenColor]; [self.view addSubview:btn];

    //зҰҒжӯўиҮӘеҠЁиҪ¬жҚўAutoresizingMask btn.translatesAutoresizingMaskIntoConstraints = NO;

    //еұ…дёӯ [self.view addConstraint:[NSLayoutConstraint constraintWithItem:btn attribute:NSLayoutAttributeCenterX relatedBy:NSLayoutRelationEqual toItem:self.view attribute:NSLayoutAttributeCenterX multiplier:1 constant:0]];

    //и·қзҰ»еә•йғЁ20еҚ•дҪҚ //жіЁж„ҸNSLayoutConstraintеҲӣе»әзҡ„constantжҳҜеҠ еңЁtoItemеҸӮж•°зҡ„пјҢжүҖд»ҘйңҖиҰҒ-20гҖӮ [self.view addConstraint:[NSLayoutConstraint constraintWithItem:btn attribute:NSLayoutAttributeBottom relatedBy:NSLayoutRelationEqual toItem:self.view attribute:NSLayoutAttributeBottom multiplier:1 constant:-20]];

    //е®ҡд№үй«ҳеәҰжҳҜзҲ¶Viewзҡ„дёүеҲҶд№ӢдёҖ [self.view addConstraint:[NSLayoutConstraint constraintWithItem:btn attribute:NSLayoutAttributeHeight relatedBy:NSLayoutRelationEqual toItem:self.view attribute:NSLayoutAttributeHeight multiplier:0.3 constant:0]];

    //жіЁеҶҢKVOж–№жі• [btn addObserver:self forKeyPath:@“bounds” options:NSKeyValueObservingOptionNew | NSKeyValueObservingOptionInitial context:nil];
    }

//KVOеӣһи°ғ

  • (void)observeValueForKeyPath:(NSString *)keyPath ofObject:(id)object change:(NSDictionary *)change context:(void *)context { if (object == btn && [keyPath isEqualToString:@“bounds”]) { [btn setTitle:NSStringFromCGSize(btn.bounds.size) forState:UIControlStateNormal]; } } {%endcodeblock%}
    иҝҗиЎҢз»“жһңпјҡ

йңҖжұӮ 2пјҡ

  1. еңЁжЁӘеҗ‘зҡ„жҳҫзӨәдёӯпјҢButtonзҡ„й«ҳеәҰеҸӘжңү96пјҢжүҖд»ҘиҰҒжұӮButtonзҡ„жңҖе°Ҹй«ҳеәҰдёә150гҖӮ

ж¶үеҸҠеҲ°зҡ„зӣёе…ізү№жҖ§пјҡ

  • дјҳе…Ҳзә§пјҡеҪ“дёӨдёӘ**ConstraintеҗҢж—¶дҪңз”ЁеңЁдёҖдёӘжҺ§д»¶ж—¶пјҢеңЁжҹҗдәӣжғ…еҶөдёӢжҳҜжңүеҶІзӘҒзҡ„пјҢеҸҜд»ҘйҖҡиҝҮи®ҫзҪ®Constraintзҡ„дјҳе…Ҳзә§жқҘи§ЈеҶігҖӮ дјҳе…Ҳзә§еҜ№еә”NSLayoutConstraintзұ»еһӢзҡ„priorityеұһжҖ§пјҢй»ҳи®ӨеҖјжҳҜUILayoutPriorityRequiredпјҢж•°еҖјдёҠзӯүдәҺ1000**. и®ҫзҪ®дёҖдёӘдҪҺзҡ„еҖјд»ЈиЎЁжӣҙдҪҺзҡ„дјҳе…Ҳзә§гҖӮ
  • жңҖе°ҸеҖјзҡ„е®ҡд№үпјҡдҪҝз”Ё**NSLayoutRelationGreaterThanOrEqualдҪңдёәNSLayoutConstraintзұ»еһӢеҲӣе»әж—¶зҡ„relatedBy**еҸӮж•°гҖӮ

дҝ®ж”№дёҠйқўзҡ„жҜ”дҫӢConstraintпјҢе№¶еңЁдёӢж–№еҠ е…ҘдёҖдёӘж–°зҡ„йҷҗеҲ¶жңҖе°ҸеҖјзҡ„ConstraintпјҢд»Јз Ғпјҡ
{%codeblock lang:objc%} //е®ҡд№үй«ҳеәҰжҳҜзҲ¶Viewзҡ„дёүеҲҶд№ӢдёҖ //и®ҫзҪ®дјҳе…Ҳзә§дҪҺдәҺUILayoutPriorityRequired(1000)пјҢUILayoutPriorityDefaultHighжҳҜ750 NSLayoutConstraint *con = [NSLayoutConstraint constraintWithItem:btn attribute:NSLayoutAttributeHeight relatedBy:NSLayoutRelationEqual toItem:self.view attribute:NSLayoutAttributeHeight multiplier:0.3 constant:0]; con.priority = UILayoutPriorityDefaultHigh; [self.view addConstraint:con];

//и®ҫзҪ®btnжңҖе°Ҹй«ҳеәҰдёә150 [btn addConstraint:[NSLayoutConstraint constraintWithItem:btn attribute:NSLayoutAttributeHeight relatedBy:NSLayoutRelationGreaterThanOrEqual toItem:nil attribute:NSLayoutAttributeNotAnAttribute multiplier:1 constant:150]]; {%endcodeblock%}
иҝҗиЎҢеҗҺпјҢжЁӘеҗ‘еұҸе№•дёӯзҡ„Buttonй«ҳеәҰжҲҗдәҶ150пјҡ

intrinsicContentSize жҺ§д»¶зҡ„еҶ…зҪ®еӨ§е°Ҹ

жҺ§д»¶зҡ„еҶ…зҪ®еӨ§е°ҸжҳҜз”ұжҺ§д»¶жң¬иә«зҡ„еҶ…е®№жүҖеҶіе®ҡзҡ„пјҢжҜ”еҰӮдёҖдёӘUILabelзҡ„ж–Үеӯ—еҫҲй•ҝпјҢйӮЈд№ҲиҜҘUILabelзҡ„еҶ…зҪ®еӨ§е°ҸиҮӘз„¶дјҡеҫҲй•ҝгҖӮ
еңЁд»Јз ҒдёӯиҺ·еҸ–жҺ§д»¶зҡ„еҶ…зҪ®еӨ§е°Ҹзҡ„ж–№жі•пјҡ

  1. йҖҡиҝҮUIViewзҡ„intrinsicContentSizeеұһжҖ§жқҘиҺ·еҸ–пјӣ
  2. йҖҡиҝҮinvalidateIntrinsicContentSizeж–№жі•жқҘеңЁдёӢж¬ЎUIи§„еҲ’дәӢд»¶дёӯйҮҚж–°и®Ўз®—intrinsicContentSizeгҖӮ

жіЁж„ҸпјҡеҰӮжһңзӣҙжҺҘеҲӣе»әдёҖдёӘеҺҹе§Ӣзҡ„UIViewеҜ№иұЎпјҢе®ғзҡ„еҶ…зҪ®еӨ§е°Ҹдёә0гҖӮ

е…ҲеҶҷдёҖдёӘиҫ…еҠ©ж–№жі•жқҘеҝ«йҖҹи®ҫзҪ®UIViewзҡ„иҫ№и·қйҷҗеҲ¶пјҡ
{%codeblock lang:objc%} //и®ҫзҪ®Autolayoutдёӯзҡ„иҫ№и·қиҫ…еҠ©ж–№жі•

  • (void)setEdge:(UIView*)superview view:(UIView*)view attr:(NSLayoutAttribute)attr constant:(CGFloat)constant { [superview addConstraint:[NSLayoutConstraint constraintWithItem:view attribute:attr relatedBy:NSLayoutRelationEqual toItem:superview attribute:attr multiplier:1.0 constant:constant] ]; } {%endcodeblock%}
    жҺҘдёӢжқҘпјҢеҲӣе»әдёҖдёӘUIViewпјҢеҲ©з”ЁдёҠйқўзҡ„иҫ…еҠ©ж–№жі•еҝ«йҖҹи®ҫзҪ®е…¶еңЁзҲ¶жҺ§д»¶зҡ„е·ҰпјҢдёҠпјҢеҸіиҫ№и·қдёә20еҚ•дҪҚгҖӮеҰӮдёӢд»Јз Ғпјҡ
    {%codeblock lang:objc%} //view1 UIView view1 = [UIView new]; view1.backgroundColor = [UIColor yellowColor]; //дёҚе…Ғи®ёAutoresizingMaskиҪ¬жҚўжҲҗAutolayout view1.translatesAutoresizingMaskIntoConstraints = NO; [self.view addSubview:view1]; //и®ҫзҪ®е·ҰпјҢдёҠпјҢеҸіиҫ№и·қдёә20. [self setEdge:self.view view:view1 attr:NSLayoutAttributeLeft constant:20]; [self setEdge:self.view view:view1 attr:NSLayoutAttributeTop constant:20]; [self setEdge:self.view view:view1 attr:NSLayoutAttributeRight constant:-20]; {%endcodeblock%}
    дҪҶжҳҜиҝҗиЎҢеҗҺдјҡеҸ‘зҺ°пјҢз•ҢйқўдёҠдёҚдјҡжҳҫзӨәд»»дҪ•дёңиҘҝгҖӮеҺҹеӣ е°ұжҳҜдёҠйқўи®Ізҡ„пјҢUIViewй»ҳи®ӨжҳҜжІЎжңү    *intrinsicContentSize**зҡ„гҖӮ

еҲӣе»әдёҖдёӘиҮӘе®ҡд№үзҡ„UIViewжқҘж”№еҶҷ**intrinsicContentSizeпјҡMyView:
з„¶еҗҺеңЁ.mж–Үд»¶дёӯж”№еҶҷintrinsicContentSize**ж–№жі•пјҢе№¶иҝ”еӣһжңүж•ҲеҖјпјҢжҜ”еҰӮиҝҷж ·пјҡ
{%codeblock lang:objc%} //ж”№еҶҷUIViewзҡ„intrinsicContentSize

  • (CGSize)intrinsicContentSize { return CGSizeMake(70, 40); }
    {%endcodeblock%}
    жҺҘзқҖдҝ®ж”№жңҖдёҠйқўзҡ„д»Јз ҒпјҢжҠҠдёҠйқўview1еҸҳйҮҸзҡ„зұ»еһӢд»ҺUIViewжӣҝжҚўжҲҗжҲ‘д»¬иҮӘе®ҡд№үзҡ„ViewпјҡMyViewзұ»еһӢпјҡ
    {%codeblock lang:objc%} MyView *view1 = [MyView new];
    {%endcodeblock%}
    еҶҚж¬ЎиҝҗиЎҢд»Јз ҒпјҢViewдјҡжҢүз…§иҰҒжұӮжҳҫзӨәеңЁеұҸе№•дёҠпјҡ
    жҢүз…§еҗҢж ·зҡ„ж–№ејҸпјҢеңЁдёӢж–№ж·»еҠ еҸҰдёҖдёӘMyViewпјҢиҰҒжұӮе…¶и·қзҰ»зҲ¶жҺ§д»¶иҫ№и·қе·ҰпјҢдёӢпјҢеҸіеҗ„дёә20пјҢд»Јз Ғпјҡ
    {%codeblock lang:objc%} //view2
    MyView *view2 = [MyView new];
    view2.backgroundColor = [UIColor yellowColor];
    //дёҚе…Ғи®ёAutoresizingMaskиҪ¬жҚўжҲҗAutolayout
    view2.translatesAutoresizingMaskIntoConstraints = NO;
    [self.view addSubview:view2];
    //и®ҫзҪ®е·ҰпјҢдёӢпјҢеҸіиҫ№и·қдёә20.
    [self setEdge:self.view view:view2 attr:NSLayoutAttributeLeft constant:20];
    [self setEdge:self.view view:view2 attr:NSLayoutAttributeBottom constant:-20];
    [self setEdge:self.view view:view2 attr:NSLayoutAttributeRight constant:-20];
    {%endcodeblock%}

йңҖжұӮпјҡ

  1. йҖҡиҝҮд»Јз ҒеҠ е…ҘAutolayoutдёӯзҡ„й—ҙи·қпјҢе‘Ҫд»Өview1е’Ңview2дёҠдёӢеҝ…йЎ»й—ҙйҡ”20дёӘеҚ•дҪҚгҖӮ 2. - иҝҷйҮҢиҰҒжұӮview2еңЁview1д№ӢдёӢзҡ„20еҚ•дҪҚпјҢжүҖд»ҘеҲӣе»әNSLayoutConstraintдёӯview2еҸӮж•°еңЁеүҚйқўгҖӮ
    2. - view2зҡ„**attributeеҸӮж•°жҳҜNSLayoutAttributeTopпјҢиҖҢview1зҡ„attributeеҸӮж•°жҳҜNSLayoutAttributeBottom**

  2. жӢүдјёview2,иҖҢдёҚжӢүдјёview1гҖӮ

      • жҺ§д»¶зҡ„**Content Hugging Priority**жӢ’з»қжӢүдјёзҡ„дјҳе…Ҳзә§пјҢдјҳе…Ҳзә§и¶Ҡй«ҳпјҢжҺ§д»¶дјҡи¶ҠдёҚе®№жҳ“иў«жӢүдјёгҖӮ
      • жҺ§д»¶зҡ„**Content Compression Resistance Priority**жӢ’з»қеҺӢзј©еҶ…зҪ®з©әй—ҙ(intrinsicContentSize)зҡ„дјҳе…Ҳзә§гҖӮдјҳе…Ҳзә§и¶Ҡй«ҳпјҢжҺ§д»¶зҡ„еҶ…зҪ®з©әй—ҙ(intrinsicContentSize)дјҡи¶ҠдёҚе®№жҳ“иў«еҺӢзј©гҖӮ

{%codeblock lang:objc%} //и®ҫзҪ®дёӨдёӘViewдёҠдёӢй—ҙи·қдёә20 [self.view addConstraint:[NSLayoutConstraint constraintWithItem:view2 attribute:NSLayoutAttributeTop relatedBy:NSLayoutRelationEqual toItem:view1 attribute:NSLayoutAttributeBottom multiplier:1.0 constant:20] ]; {%endcodeblock%}

OKпјҢзҡ„зЎ®пјҢжӯӨж—¶view1е’Ңview2зӣёдә’й—ҙйҡ”20еҚ•дҪҚпјҢдҪҶжҳҜview1иў«жӢүдјёдәҶгҖӮ

дҪҝз”ЁжҺ§д»¶зҡ„**Content Hugging PriorityпјҢеҰӮдёӢеӣҫпјҡ
еҰӮеӣҫ,жҠҠview1пјҲдёҠеӣҫдёӯиў«жӢүдјёзҡ„пјҢеңЁдёҠйқўзҡ„Viewпјүзҡ„Content Hugging Priorityи®ҫзҪ®дёҖдёӘжӣҙй«ҳзҡ„еҖјпјҢйӮЈд№ҲеҪ“AutolayoutйҒҮеҲ°иҝҷз§ҚеҶіе®ҡи°ҒжқҘжӢүдјёзҡ„жғ…еҶөж—¶пјҢview1дёҚдјҡиў«дјҳе…ҲжӢүдјёпјҢиҖҢдјҳе…Ҳзә§зЁҚдҪҺзҡ„view2жүҚдјҡиў«жӢүдјёгҖӮ
еҸҜд»ҘзӣҙжҺҘйҖҡиҝҮUIViewзҡ„setContentHuggingPriority:forAxisж–№жі•жқҘи®ҫзҪ®жҺ§д»¶зҡ„Content Hugging PriorityпјҢе…¶дёӯforAxisеҸӮж•°д»ЈиЎЁжЁӘеҗ‘е’Ңзәөеҗ‘пјҢжң¬дҫӢдёӯеҸӘйңҖиҰҒи®ҫзҪ®зәөеҗ‘пјҢжүҖд»Ҙдј е…ҘUILayoutConstraintAxisVertical**гҖӮж•ҙеҸҘд»Јз Ғпјҡ
{%codeblock lang:objc%} //жҸҗй«ҳview1зҡ„Content Hugging Priority [view1 setContentHuggingPriority:UILayoutPriorityDefaultHigh forAxis:UILayoutConstraintAxisVertical]; {%endcodeblock%}
